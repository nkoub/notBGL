763 template<typename graph_t>
764 auto notBGL::load_edgelist(std::string filename, graph_t &g)
765 {
766  // Vertex descriptors.
767  typedef typename boost::graph_traits<graph_t>::vertex_descriptor vertex_t;
768  vertex_t v1, v2;
769  // Stream objects.
770  std::ifstream edgelist_file;
771  std::stringstream one_line;
772  // String objects.
773  std::string full_line, name1_str, name2_str;
774  // Integer objects.
775  unsigned int node_cnt(0);
776  // Map objects to assure the uniqueness of vertices.
777  std::map< std::string, vertex_t > Name2Vertex;
778  // Iterator objects.
779  typename std::map< std::string, vertex_t >::iterator name_it;
780  // Opens the stream and terminates if the operation did not succeed.
781  edgelist_file.open(filename.c_str(), std::ios_base::in);
782  if( !edgelist_file.is_open() )
783  {
784  std::cerr << "Could not open file: " << filename << "." << std::endl;
785  std::terminate();
786  }
787  else
788  {
789  // Reads the edgelist and registers the nodes and the edges.
790  while( !edgelist_file.eof() )
791  {
792  // Reads a line of the edgelist.
793  std::getline(edgelist_file,full_line); edgelist_file >> std::ws;
794  one_line.str(full_line); one_line >> std::ws;
795  one_line >> name1_str >> std::ws;
796  // Skips a line of comment.
797  if(name1_str == "#")
798  {
799  one_line.clear();
800  continue;
801  }
802  one_line >> name2_str >> std::ws;
803  one_line.clear();
804  // Is name1 new?
805  name_it = Name2Vertex.find(name1_str);
806  if( name_it == Name2Vertex.end() )
807  {
808  v1 = boost::add_vertex(g);
809  Name2Vertex[name1_str] = v1;
810  g[v1].name = name1_str;
811  g[v1].num = node_cnt;
812  ++node_cnt;
813  }
814  else
815  {
816  v1 = name_it->second;
817  }
818  // Is name2 new?
819  name_it = Name2Vertex.find(name2_str);
820  if( name_it == Name2Vertex.end() )
821  {
822  v2 = boost::add_vertex(g);
823  Name2Vertex[name2_str] = v2;
824  g[v2].name = name2_str;
825  g[v2].num = node_cnt;
826  ++node_cnt;
827  }
828  else
829  {
830  v2 = name_it->second;
831  }
832  // Creates the edge if it does not exist (forbids multiple edges and self-loops).
833  if(v1 != v2) // not a self-loop.
834  {
835  if(boost::edge(v1,v2,g).second == false) // not a multiple edge
836  {
837  boost::add_edge(v1, v2, g);
838  }
839  }
840  }
841  }
842  // Closes the stream.
843  edgelist_file.close();
844  // Returns the maps "Name2Vertex" to help add other properties to vertices.
845  return Name2Vertex;
846 }

1634 template<typename graph_t>
1635 auto notBGL::kcore_decomposition(graph_t &graph)
1636 {
1637  // Typedef of a vertex.
1638  typedef typename graph_t::vertex_descriptor vertex_t;
1639  // std::map object containing the coreness of each vertex.
1640  typename std::map<vertex_t, double> Vertex2kCore;
1641  // std::vector object containing the effective degree of each vertex.
1642  std::vector<unsigned int> DegreeVec(boost::num_vertices(graph));
1643  // Builds the list of the degree of the vertices.
1644  typename graph_t::vertex_iterator v_it, v_end;
1645  typename graph_t::adjacency_iterator a_it, a_end;
1646  std::set<std::pair<unsigned int, vertex_t> > DegreeSet;
1647  for (std::tie(v_it, v_end) = boost::vertices(graph); v_it!=v_end; ++v_it)
1648  {
1649  DegreeSet.insert(std::make_pair(boost::out_degree(*v_it, graph), *v_it));
1650  DegreeVec[graph[*v_it].num] = boost::out_degree(*v_it, graph);
1651  }
1652  // Sets the coreness of each vertex using the algorithm by Batagelj and Zaversnik.
1653  vertex_t v1, v2;
1654  unsigned int d1, d2, n2;
1655  typename std::set<std::pair<unsigned int, vertex_t> >::iterator m_it;
1656  while(DegreeSet.size() > 0)
1657  {
1658  // Sets the coreness of the first vertex of the list and removes it from the list.
1659  d1 = DegreeSet.begin()->first;
1660  v1 = DegreeSet.begin()->second;
1661  Vertex2kCore[v1] = d1;
1662  DegreeSet.erase(DegreeSet.begin());
1663  // Reduces the "effective" degree of its neighbours.
1664  for(std::tie(a_it, a_end) = boost::adjacent_vertices(v1, graph); a_it!=a_end; ++a_it)
1665  {
1666  n2 = graph[*a_it].num;
1667  d2 = DegreeVec[n2];
1668  m_it = DegreeSet.find(std::make_pair(d2, *a_it));
1669  if(m_it != DegreeSet.end())
1670  {
1671  // std::cout << graph[m_it->second].name << std::endl;
1672  d2 = m_it->first;
1673  if(d2 > d1)
1674  {
1675  v2 = m_it->second;
1676  DegreeVec[n2] = d2 - 1;
1677  DegreeSet.erase(m_it);
1678  DegreeSet.insert(std::make_pair(d2 - 1, v2));
1679  }
1680  }
1681  }
1682  }
1683  // Returns the coreness of the vertices.
1684  return Vertex2kCore;
1685 }
