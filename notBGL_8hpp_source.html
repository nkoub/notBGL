537 template<typename graph_t>
538 auto notBGL::load_edgelist(std::string filename, graph_t &g)
539 {
540  // Vertex descriptors.
541  typedef typename boost::graph_traits<graph_t>::vertex_descriptor vertex_t;
542  vertex_t v1, v2;
543  // Stream objects.
544  std::ifstream edgelist_file;
545  std::stringstream one_line;
546  // String objects.
547  std::string full_line, name1_str, name2_str;
548  // Integer objects.
549  unsigned int node_cnt(0);
550  // Map objects to assure the uniqueness of vertices.
551  std::map< std::string, vertex_t > Name2Vertex;
552  // Iterator objects.
553  typename std::map< std::string, vertex_t >::iterator name_it;
554  // Opens the stream and terminates if the operation did not succeed.
555  edgelist_file.open(filename.c_str(), std::ios_base::in);
556  if( !edgelist_file.is_open() )
557  {
558  std::cerr << "Could not open file: " << filename << "." << std::endl;
559  std::terminate();
560  }
561  else
562  {
563  // Reads the edgelist and registers the nodes and the edges.
564  while( !edgelist_file.eof() )
565  {
566  // Reads a line of the edgelist.
567  std::getline(edgelist_file,full_line); edgelist_file >> std::ws;
568  one_line.str(full_line); one_line >> std::ws;
569  one_line >> name1_str >> std::ws;
570  one_line >> name2_str >> std::ws;
571  one_line.clear();
572  // Is name1 new?
573  name_it = Name2Vertex.find(name1_str);
574  if( name_it == Name2Vertex.end() )
575  {
576  v1 = boost::add_vertex(g);
577  Name2Vertex[name1_str] = v1;
578  g[v1].name = name1_str;
579  g[v1].id = node_cnt;
580  ++node_cnt;
581  }
582  else
583  {
584  v1 = name_it->second;
585  }
586  // Is name2 new?
587  name_it = Name2Vertex.find(name2_str);
588  if( name_it == Name2Vertex.end() )
589  {
590  v2 = boost::add_vertex(g);
591  Name2Vertex[name2_str] = v2;
592  g[v2].name = name2_str;
593  g[v2].id = node_cnt;
594  ++node_cnt;
595  }
596  else
597  {
598  v2 = name_it->second;
599  }
600  // Creates the edge if it does not exist (forbids multiple edges and self-loops).
601  if(v1 != v2) // not a self-loop.
602  {
603  if(boost::edge(v1,v2,g).second == false) // not a multiple edge
604  {
605  boost::add_edge(v1, v2, g);
606  }
607  }
608  }
609  }
610  // Closes the stream.
611  edgelist_file.close();
612  // Returns the maps "Name2Vertex" to help add other properties to vertices.
613  return Name2Vertex;
614 }

837 template<typename graph_t>
838 auto notBGL::survey_triangles(graph_t &g)
839 {
840  // Typedef.
841  typedef typename boost::graph_traits<graph_t>::vertex_descriptor vertex_t;
842  // typedef typename boost::graph_traits<graph_t>::edge_descriptor edge_t;
843  // Vector objects.
844  std::vector<vertex_t> intersection;
845  std::vector<std::vector<vertex_t> > triangles;
846  // std::vector<std::vector<vertex_t> > triangles;
847  // Set objects.
848  std::set<vertex_t> neighbours_v1, neighbours_v2;
849  // Iterator objects.
850  typename std::vector<vertex_t>::iterator it;
851  typename boost::graph_traits<graph_t>::vertex_iterator it1, end1;
852  typename boost::graph_traits<graph_t>::adjacency_iterator it2, end2, it3, end3;
853  // Vertex descriptor objects.
854  vertex_t v3;
855  // Objects containing the degree of vertices.
856  typename boost::graph_traits<graph_t>::degree_size_type d1, d2;
857  // Computes the intersection for the in- and out- neighbourhoods of each node.
858  for(std::tie(it1, end1) = vertices(g); it1 != end1; ++it1)
859  {
860  // Degree of vertex v1.
861  d1 = out_degree(*it1, g);
862  // Performs the calculation only if d1>1.
863  if( d1 > 1 )
864  {
865  // Builds an ordered list of the neighbourhood of v1
866  neighbours_v1.clear();
867  std::tie(it2, end2) = adjacent_vertices(*it1, g);
868  neighbours_v1.insert(it2, end2);
869  // Loops over the neighbours of vertex v1.
870  for(std::tie(it2, end2) = adjacent_vertices(*it1, g); it2!=end2; ++it2)
871  {
872  // Identity and degree of vertex 2.
873  d2 = out_degree(*it2, g);
874  // Performs the calculation only if d1>1 and if v2>v1 (ensures that each triangle is counted once).
875  if( *it1 < *it2 && d2 > 1 )
876  {
877  // Builds an ordered list of the neighbourhood of v2
878  neighbours_v2.clear();
879  for(std::tie(it3, end3) = adjacent_vertices(*it2, g); it3 != end3; ++it3)
880  {
881  if(*it2 < *it3) // Ensures that triangles will be counted only once.
882  {
883  neighbours_v2.insert(*it3);
884  }
885  }
886  // Identifies the triangles.
887  intersection.clear();
888  intersection.resize(std::min(d1, neighbours_v2.size()));
889  it = std::set_intersection(neighbours_v1.begin(), neighbours_v1.end(), neighbours_v2.begin(), neighbours_v2.end(), intersection.begin());
890  intersection.resize(it-intersection.begin());
891  // Loops over the common neighbours of vertices v1 and v2.
892  for(unsigned int n(0), nn(intersection.size()); n<nn; ++n)
893  {
894  // Identity and degree of vertex 2.
895  v3 = intersection[n];
896  // Adds the triangle to the list.
897  triangles.push_back({*it1, *it2, intersection[n]});
898  }
899  }
900  }
901  }
902  }
903  // Returns the vector containing the triangles.
904  return triangles;
905 }

925  {

927  std::tie(it, inserted) = local_clustering_coefficient.insert(std::make_pair(v, double(0)));

930  it->second = 1;
931  }
932  else
933  {
934  it->second += 1;
935  }
936  }
937  }

951  }
952  }
